const boxes = [...document.querySelectorAll('.box')];
function boxClick(e) {
  const idxBox = boxes.indexOf(e.target);
  boxes[idxBox].textContent = letter;
  letter = (letter == 'X') ? 'O' : 'X';
}
boxes.forEach(box => box.addEventListener('click', boxClick));
