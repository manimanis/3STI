Structures conditionnelles

Structure if

Exemple

// Comparaison de deux entiers a et b
if (a > b) {
  res = a + " > " + b;
} else if (a < b) {
  res = a + " < " + b;
} else {
  res = a + " = " + b;
}

Application

Ecrire un script qui saisit trois nombres distincts a, b et c puis affiche l'entier qui sera au milieu si les trois entiers ont été ordonnés.

Exemple : a = 5, b = 16 et c = 10, le nombre au milieu est 10 car 5 < 10 < 16

let a = Number(prompt("Donner a ?", "5"));
let b = Number(prompt("Donner b ?", "16"));
let c = Number(prompt("Donner c ?", "10"));

let mid;
if ((a > b && b > c) || (a < b && b < c)) {
  mid = b;
} else if ((b > a && a > c) || (b < a && a < c)) {
  mid = a;
} else {
  mid = c;
}

alert("Nombre au milieu : " + mid);

Application

Un nombre est dit premier si et seulement s'il est divisible exactement par deux nombres distincts 1 et lui même. Ecrire un script qui affiche tous les nombres premiers inférieurs à 100.

document.writeln("<p>Nombre premiers < 100 :");
for (let i = 0; i < 100; i++) {
  let nbd = 0;
  for (let j = 1; j <= i; j++) {
    if (i % j == 0) {
      nbd++;
    }
  }
  if (nbd == 2) {
    document.writeln(i, ", ");
  }
}
document.writeln("</p>");

Instruction while

Exemple

// Déterminer si une chaîne est palindrôme
// Exemples : AZIZA, RADAR, RESSASSER
let ch = "RESSASSER";
let i = 0, j = ch.length - 1;
while (j > i && ch[i] == ch[j]) {
  i++; j--;
}
document.writeln("<p>", ch, " est ", (i >= j ? "palindrôme" : "non palindrôme"), "</p>");

Application

Deux nombres a et b sont premiers entre eux si et seulement si PGCD(a, b) = 1. Ecrire un script qui détermine si deux entiers sont premiers entre eux.

PGCD(a, b) = PGCD(b, a mod b) ; si b ≠ 0
PGCD(a, b) = a ; si b = 0

let a = Number(prompt("Donner a ? ", "35"));
let b = Number(prompt("Donner b ? ", "12"));
let a1 = a, b1 = b;
while (b1 != 0) {
  let r = a1 % b1;
  a1 = b1;
  b1 = r;
}
document.writeln("<p>", a, " et ", b, " ", 
  (a1 == 1 ? "sont" : "ne sont pas"), 
  " premiers entre eux.</p>");

  showMessage('Vous avez cliqué sur le premier bouton');

}

// Définition de la fonction qui gère l'évènement submit sur le formulaire
function handleFormSubmit(event) {
  // Annulation de l'envoi du formulaire
  event.preventDefault();

  // Récupération de la valeur du champ de formulaire
  var inputValue = nameInput.value;

  // Affichage de la valeur du champ de formulaire dans le paragraphe
  showMessage('Vous vous appelez ' + inputValue);
}

form.addEventListener('submit', handleFormSubmit);
